class User:
    def __init__(self, name):
        self.name = name
        self.age = None 

    def get_name(self):
        return self.name

    def set_name(self, name):
        self.name = name

    def get_age(self):
        return self.age

    def set_age(self, age):
        self.age = age


class Employee(User):
    def __init__(self, name):
        super().__init__(name)
        self._age = None 

    def get_age(self):
        return self._age

    def set_age(self, age):
        if 18 <= age <= 65:
            self._age = age
        else:
            raise ValueError("Возраст должен быть от 18 до 65 лет")

employee = Employee("John")
employee.set_age(30)
print(f"Возраст: {employee.get_age()}")

try:
    employee.set_age(10)
except ValueError as e:
    print(f"Ошибка: {e}")

try:
  employee.set_age(70)
except ValueError as e:
  print(f"Ошибка: {e}")
