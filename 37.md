```python
class User:
    def setName(self, name):
        if self._notEmpty(name):  # вызов защищенного метода
            self.name = name

    def getName(self):
        return self.name

    def _notEmpty(self, stri):
        return len(stri) > 0  # метод защищен, по соглашению, и должен иметь один подчеркивание
        

class Employee(User):
    def setSurn(self, surname):
        if self._notEmpty(surname):
            self.surname = surname

    def getSurn(self):
        return self.surname
```
